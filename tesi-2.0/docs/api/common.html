1957def _no_init_or_replace_init(self, *args, **kwargs):
1958    cls = type(self)
1959
1960    if cls._is_protocol:
1961        raise TypeError('Protocols cannot be instantiated')
1962
1963    # Already using a custom `__init__`. No need to calculate correct
1964    # `__init__` to call. This can lead to RecursionError. See bpo-45121.
1965    if cls.__init__ is not _no_init_or_replace_init:
1966        return
1967
1968    # Initially, `__init__` of a protocol subclass is set to `_no_init_or_replace_init`.
1969    # The first instantiation of the subclass will call `_no_init_or_replace_init` which
1970    # searches for a proper new `__init__` in the MRO. The new `__init__`
1971    # replaces the subclass' old `__init__` (ie `_no_init_or_replace_init`). Subsequent
1972    # instantiation of the protocol subclass will thus use the new
1973    # `__init__` and no longer call `_no_init_or_replace_init`.
1974    for base in cls.__mro__:
1975        init = base.__dict__.get('__init__', _no_init_or_replace_init)
1976        if init is not _no_init_or_replace_init:
1977            cls.__init__ = init
1978            break
1979    else:
1980        # should not happen
1981        cls.__init__ = object.__init__
1982
1983    cls.__init__(self, *args, **kwargs)

35def update_status(queue_path: Path, sprint_id: str, **updates) -> bool:
36    """Aggiorna in-place lo status di uno sprint (read-modify-write atomico via lock)."""
37    with queue_lock(queue_path, "r+") as f:
38        lines = f.read().splitlines()
39        f.seek(0)
40        f.truncate()
41        found = False
42        for ln in lines:
43            if not ln.strip():
44                continue
45            e = json.loads(ln)
46            if e["id"] == sprint_id:
47                e.update(updates)
48                found = True
49            f.write(json.dumps(e, ensure_ascii=False) + "\n")
50    if found:
51        from common.broadcast import emit
52
53        if updates.get("status") == "claimed":
54            emit("sprint_claimed", sprint_id=sprint_id, owner=updates.get("owner"))
55        if updates.get("status") == "done":
56            emit("sprint_done", sprint_id=sprint_id, verdict=updates.get("verdict"))

134    """Applica la migrazione ``version`` su ``target``. Idempotente.
135
136    Returns:
137        dict con ``version``, ``status`` (applied|skipped), ``records_affected``.
138    """
139    _load_migration_modules()
140    if version not in _REGISTRY:
141        raise KeyError(f"Migration {version!r} not registered.")
142    if _is_applied(target, version):
143        return {"version": version, "status": "skipped", "records_affected": 0}
144
145    entry = _REGISTRY[version]
146    records = _read_records(target)
147    transformed = entry.forward_fn(list(records))
148    _write_records(target, transformed)
149
150    state = _load_state(target)
151    state.setdefault("applied", []).append({
152        "version": version,
153        "direction": "forward",
154        "ts": datetime.now(timezone.utc).isoformat(),
155        "records_before": len(records),
156        "records_after": len(transformed),
157    })
158    _save_state(target, state)
159    return {"version": version, "status": "applied", "records_affected": len(transformed)}

def register_migration(version: str, description: str = '') -> Callable: 
41def register_migration(version: str, description: str = "") -> Callable:
42    """Decorator per registrare una migrazione nel registry globale.
43
44    La classe decorata deve definire:
45      - ``forward(records: list[dict]) -> list[dict]``
46      - ``backward(records: list[dict]) -> list[dict]``
47    """
48    def decorator(cls):
49        if version in _REGISTRY:
50            raise ValueError(f"Migration version already registered: {version!r}")
51        if not (hasattr(cls, "forward") and hasattr(cls, "backward")):
52            raise TypeError(f"{cls.__name__} must define forward() and backward()")
53        _REGISTRY[version] = _MigrationEntry(
54            version=version,
55            description=description,
56            forward_fn=cls.forward,
57            backward_fn=cls.backward,
58        )
59        return cls
60    return decorator
