 81    """Salva la lista di STC su disco.
 82
 83    Parameters
 84    ----------
 85    stcs:
 86        Lista di SourceEstimate da salvare.
 87    out_dir:
 88        Directory di output (creata se non esiste).

 90        Prefisso per i file (es. 'sub-05_rs-epoch').
 91

 97    out_dir.mkdir(parents=True, exist_ok=True)
 98    saved: list[Path] = []
 99    for idx, stc in enumerate(stcs):
100        stem = out_dir / f"{prefix}_epoch-{idx:04d}_inv"
101        stc.save(str(stem), overwrite=True)
102        saved.append(stem)
103    return saved
